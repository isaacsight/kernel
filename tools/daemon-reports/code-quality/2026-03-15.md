**Recommendations:**

   ```typescript

     Free = 'free',
     ProMonthly = 'pro_monthly'
   }
   ```

2. **Update `resolvePlanId` to Use the Enum and Validate Plans:**
   ```typescript
   function resolvePlanId(sub: { status: string; plan?: string } | null): PlanType {
     if (!sub || !['active', 'trialing', 'past_due'].includes(sub.status)) return PlanType.Free;
     
     const allowedPlans = [PlanType.Free, PlanType.ProMonthly];
     const plan = sub.plan ? (allowedPlans.find(p => p.toString() === sub.plan) ?? PlanType.Free);
     return plan;
   }
   ```
